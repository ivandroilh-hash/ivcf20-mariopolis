25       fun calcularIdade(): Int {
26           if (dataNasc.isEmpty()) return 0
27           return try {
28               val sdf = SimpleDateFormat("dd/MM/yyyy", Locale.getDefault())
29               val dataNascimento = sdf.parse(dataNasc) ?: return 0
30               val hoje = Calendar.getInstance()
31               val nascimento = Calendar.getInstance().apply { time = dataNascimento }
32               var idade = hoje.get(Calendar.YEAR) - nascimento.get(Calendar.YEAR)
33               if (hoje.get(Calendar.DAY_OF_YEAR) < nascimento.get(Calendar.DAY_OF_YEAR)) idade--
34               idade
35           } catch (e: Exception) { 0 }
36       }
37   
38       fun calcularDiasVida(): Long {
39           if (dataNasc.isEmpty()) return 999
40           return try {
41               val sdf = SimpleDateFormat("dd/MM/yyyy", Locale.getDefault())
42               val dataNascimento = sdf.parse(dataNasc) ?: return 999
43               val diff = Date().time - dataNascimento.time
44               TimeUnit.DAYS.convert(diff, TimeUnit.MILLISECONDS)
45           } catch (e: Exception) { 999 }
46       }
47   }
