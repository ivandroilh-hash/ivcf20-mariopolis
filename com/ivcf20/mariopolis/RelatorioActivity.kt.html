104      private fun gerarPdfGeral(visitas: List<Visita>) {
105          val pdfDocument = PdfDocument()
106          val paint = Paint().apply { textSize = 12f }
107          val paintBold = Paint().apply { isFakeBoldText = true; textSize = 14f }
108          var pageNumber = 1
109          var pageInfo = PdfDocument.PageInfo.Builder(595, 842, pageNumber).create()
110          var page = pdfDocument.startPage(pageInfo)
111          var canvas = page.canvas
112          var y = 50f
113          val marginX = 50f
114          val maxWidth = 500f // Largura útil da página para o texto
115  
116          visitas.forEach { visita ->
117              if (y > 700) {
118                  pdfDocument.finishPage(page)
119                  pageNumber++
120                  pageInfo = PdfDocument.PageInfo.Builder(595, 842, pageNumber).create()
121                  page = pdfDocument.startPage(pageInfo)
122                  canvas = page.canvas
123                  y = 50f
124              }
125              
126              // Título (Nome)
127              canvas.drawText(visita.nome, marginX, y, paintBold)
128              y += 20f
129              
130              // Data e Hora
131              canvas.drawText("Data/Hora: ${visita.dataHora}", marginX, y, paint)
132              y += 25f
133              
134              // Conteúdo com quebra automática de linha
135              val linhas = visita.detalhamento.split("\n")
136              linhas.forEach { linha ->
137                  val wrapLinhas = wrapText(linha, paint, maxWidth)
138                  wrapLinhas.forEach { textoQuebrado ->
139                      if (y > 800) {
140                          pdfDocument.finishPage(page)
141                          pageNumber++
142                          pageInfo = PdfDocument.PageInfo.Builder(595, 842, pageNumber).create()
143                          page = pdfDocument.startPage(pageInfo)
144                          canvas = page.canvas
145                          y = 50f
146                      }
147                      canvas.drawText(textoQuebrado, marginX, y, paint)
148                      y += 15f
149                  }
150              }
151              
152              y += 10f
153              canvas.drawLine(marginX, y, 545f, y, paint)
154              y += 30f
155          }
156          pdfDocument.finishPage(page)
157  
158          val timeStamp = SimpleDateFormat("dd-MM-yyyy_HH-mm", Locale.getDefault()).format(Date())
159          val fileName = "Relatorio_Visitas_$timeStamp.pdf"
160          try {
161              if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.Q) {
162                  val contentValues = ContentValues().apply {
163                      put(MediaStore.MediaColumns.DISPLAY_NAME, fileName)
164                      put(MediaStore.MediaColumns.MIME_TYPE, "application/pdf")
165                      put(MediaStore.MediaColumns.RELATIVE_PATH, Environment.DIRECTORY_DOWNLOADS)
166                      put(MediaStore.MediaColumns.IS_PENDING, 1)
167                  }
168                  val uri = contentResolver.insert(MediaStore.Downloads.EXTERNAL_CONTENT_URI, contentValues)
169                  uri?.let {
170                      contentResolver.openOutputStream(it)?.use { pdfDocument.writeTo(it) }
171                      contentValues.clear()
172                      contentValues.put(MediaStore.MediaColumns.IS_PENDING, 0)
173                      contentResolver.update(it, contentValues, null, null)
174                  }
175              } else {
176                  val file = File(Environment.getExternalStoragePublicDirectory(Environment.DIRECTORY_DOWNLOADS), fileName)
177                  pdfDocument.writeTo(FileOutputStream(file))
178              }
179              Toast.makeText(this, "✅ Salvo!", Toast.LENGTH_LONG).show()
180          } catch (e: Exception) { Toast.makeText(this, "❌ Erro: ${e.message}", Toast.LENGTH_LONG).show() }
181          finally { pdfDocument.close() }
182      }

184      // Função auxiliar para quebrar o texto em várias linhas conforme a largura da página
185      private fun wrapText(text: String, paint: Paint, maxWidth: Float): List<String> {
186          val result = mutableListOf<String>()
187          val words = text.split(" ")
188          var line = ""
189          
190          for (word in words) {
191              val testLine = if (line.isEmpty()) word else "$line $word"
192              val width = paint.measureText(testLine)
193              if (width <= maxWidth) {
194                  line = testLine
195              } else {
196                  result.add(line)
197                  line = word
198              }
199          }
200          if (line.isNotEmpty()) result.add(line)
201          return if (result.isEmpty()) listOf("") else result
202      }
