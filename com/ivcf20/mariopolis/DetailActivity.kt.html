44       private fun formatQuestionnaireJson(jsonStr: String, cpf: String, sexo: String, nasc: String, cns: String, endereco: String): String {
45           return try {
46               val json = JSONObject(jsonStr)
47               val sb = StringBuilder()
48               
49               val vCpf = if (cpf.isNotEmpty()) cpf else json.optString("p_cpf")
50               val vSexo = if (sexo.isNotEmpty()) sexo else json.optString("p_sexo")
51               val vNasc = if (nasc.isNotEmpty()) nasc else json.optString("2")
52               val vCns = if (cns.isNotEmpty()) cns else json.optString("p_cns")
53               val vEnd = if (endereco.isNotEmpty() && endereco != ",  - ") endereco 
54                          else "${json.optString("p_rua")}, ${json.optString("p_numero")} - ${json.optString("p_bairro")}"
55   
56               sb.append("=== DADOS IDENTIFICAÇÃO ===\n")
57               sb.append("Nome: ${intent.getStringExtra("EXTRA_TITLE")}\n")
58               if (vCpf.isNotEmpty()) sb.append("CPF: $vCpf\n")
59               if (vSexo.isNotEmpty()) sb.append("Sexo: $vSexo\n")
60               if (vNasc.isNotEmpty()) sb.append("Data Nasc: $vNasc\n")
61               if (vCns.isNotEmpty()) sb.append("CNS: $vCns\n")
62               if (vEnd.length > 5) sb.append("Endereço: $vEnd\n")
63               sb.append("===========================\n\n")
64   
65               val questions = QuestionnaireActivity.questions
66               questions.forEach { q ->
67                   if (q.numero in 1..5) {
68                       val resp = json.optString(q.numero.toString())
69                       if (resp.isNotEmpty() && resp != "Sem resposta") {
70                           sb.append("${q.texto}\n")
71                           sb.append("R: $resp\n\n")
72                       }
73                   } else if (q.numero != 20) {
74                       val answer = json.optString(q.numero.toString(), "Sem resposta")
75                       if (answer != "null" && answer.isNotEmpty()) {
76                           sb.append("${q.texto}\n")
77                           sb.append("R: $answer\n\n")
78                       }
79                   }
80               }
81               sb.toString()
82           } catch (e: Exception) {
83               jsonStr
84           }
85       }
86   }
