1    package com.ivcf20.mariopolis
2    
3    import android.graphics.Color
4    import android.os.Bundle
5    import android.view.View
6    import android.view.ViewGroup
7    import android.widget.ArrayAdapter
8    import android.widget.TextView
9    import android.widget.Toast
10   import androidx.appcompat.app.AppCompatActivity
11   import com.ivcf20.mariopolis.databinding.ActivitySearchBinding
12   
13   class SearchActivity : AppCompatActivity() {
14   
15       private lateinit var binding: ActivitySearchBinding
16   
17       override fun onCreate(savedInstanceState: Bundle?) {
18           super.onCreate(savedInstanceState)
19           binding = ActivitySearchBinding.inflate(layoutInflater)
20           setContentView(binding.root)
21   
22           window.decorView.setBackgroundColor(Color.WHITE)
23   
24           setupAutoComplete()
25   
26           binding.btnSearch.setOnClickListener {
27               val query = binding.actvSearch.text.toString().trim()
28               if (query.isNotEmpty()) {
29                   realizarPesquisa(query)
30               } else {
31                   Toast.makeText(this, "Digite um nome ou CPF.", Toast.LENGTH_SHORT).show()
32               }
33           }
34       }
35   
36       private fun setupAutoComplete() {
37           val adapterSearch = object : ArrayAdapter<Imovel>(this, android.R.layout.simple_dropdown_item_1line, ImovelData.imoveis) {
38               override fun getView(position: Int, convertView: View?, parent: ViewGroup): View {
39                   val view = super.getView(position, convertView, parent) as TextView
40                   view.setTextColor(Color.BLACK)
41                   return view
42               }
43           }
44           
45           binding.actvSearch.apply {
46               setAdapter(adapterSearch)
47               setDropDownBackgroundResource(android.R.color.white)
48               setTextColor(Color.BLACK)
49               setHintTextColor(Color.GRAY)
50           }
51           
52           binding.actvSearch.setOnItemClickListener { parent, _, position, _ ->
53               val selecionado = parent.getItemAtPosition(position) as Imovel
54               mostrarDetalhesPessoa(selecionado)
55           }
56       }
57   
58       private fun realizarPesquisa(query: String) {
59           val queryLimpa = query.replace(".", "").replace("-", "").replace(" ", "")
60           
61           val resultados = ImovelData.imoveis.filter { 
62               it.nome.contains(query, ignoreCase = true) || 
63               it.cpf.replace(".", "").replace("-", "").contains(queryLimpa) ||
64               it.cns.replace(".", "").replace(" ", "").contains(queryLimpa)
65           }
66   
67           if (resultados.isEmpty()) {
68               binding.tvDetails.text = "Nenhuma pessoa encontrada."
69               binding.tvDetails.setTextColor(Color.RED)
70               return
71           }
72   
73           if (resultados.size == 1) {
74               mostrarDetalhesPessoa(resultados[0])
75           } else {
76               val sb = StringBuilder()
77               sb.append("Encontrados ${resultados.size} resultados. Selecione um na lista suspensa para ver detalhes.\n\n")
78               resultados.take(15).forEach {
79                   sb.append("- ${it.nome}\n")
80               }
81               if (resultados.size > 15) sb.append("\n... e mais ${resultados.size - 15} pessoas.")
82   
83               binding.tvDetails.text = sb.toString()
84               binding.tvDetails.setTextColor(Color.BLACK)
85           }
86       }
87   
88       private fun mostrarDetalhesPessoa(pessoa: Imovel) {
89           val detalhes = StringBuilder()
90           detalhes.append("NOME: ${pessoa.nome}\n")
91           detalhes.append("CPF: ${pessoa.cpf}\n")
92           detalhes.append("CNS: ${pessoa.cns}\n")
93           detalhes.append("DATA NASC: ${pessoa.dataNasc}\n")
94           detalhes.append("SEXO: ${pessoa.sexo}\n")
95           detalhes.append("MÃE: ${pessoa.nomedamae}\n")
96           detalhes.append("ENDEREÇO: ${pessoa.endereco}, ${pessoa.numero}\n")
97           detalhes.append("BAIRRO: ${pessoa.bairro}\n")
98           detalhes.append("ACS: ${pessoa.acs} (Área ${pessoa.area})\n")
99           detalhes.append("CÓDIGO: ${pessoa.cod}\n")
100          if (pessoa.condicoesIniciais.isNotEmpty()) {
101              detalhes.append("\nCONDIÇÕES: ${pessoa.condicoesIniciais}")
102          }
103  
104          binding.tvDetails.text = detalhes.toString()
105          binding.tvDetails.setTextColor(Color.BLACK)
106      }
107  }
