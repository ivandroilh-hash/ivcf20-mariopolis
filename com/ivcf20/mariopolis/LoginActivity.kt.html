1    package com.ivcf20.mariopolis
2    
3    import android.content.Context
4    import android.os.Bundle
5    import android.view.View
6    import android.widget.Toast
7    import androidx.appcompat.app.AppCompatActivity
8    import androidx.lifecycle.lifecycleScope
9    import com.ivcf20.mariopolis.databinding.ActivityLoginBinding
10   import com.ivcf20.mariopolis.network.SipiManager
11   import kotlinx.coroutines.launch
12   
13   class LoginActivity : AppCompatActivity() {
14   
15       private lateinit var binding: ActivityLoginBinding
16   
17       override fun onCreate(savedInstanceState: Bundle?) {
18           super.onCreate(savedInstanceState)
19           binding = ActivityLoginBinding.inflate(layoutInflater)
20           setContentView(binding.root)
21   
22           val prefs = getSharedPreferences("SipiPrefs", Context.MODE_PRIVATE)
23           binding.etEmail.setText(prefs.getString("last_email", "ivandro.ilh@gmail.com"))
24           binding.etSenha.setText(prefs.getString("last_password", "Holtz;@5122"))
25   
26           binding.btnLogin.setOnClickListener {
27               val email = binding.etEmail.text.toString().trim()
28               val senha = binding.etSenha.text.toString().trim()
29   
30               if (email.isEmpty() || senha.isEmpty()) {
31                   Toast.makeText(this, "Preencha todos os campos", Toast.LENGTH_SHORT).show()
32                   return@setOnClickListener
33               }
34   
35               realizarLogin(email, senha)
36           }
37       }
38   
39       private fun realizarLogin(email: String, senha: String) {
40           binding.btnLogin.visibility = View.GONE
41           binding.progressBar.visibility = View.VISIBLE
42   
43           lifecycleScope.launch {
44               val erro = SipiManager.autenticarComCredenciais(email, senha)
45               
46               binding.btnLogin.visibility = View.VISIBLE
47               binding.progressBar.visibility = View.GONE
48   
49               if (erro == null) {
50                   // Sucesso
51                   val prefs = getSharedPreferences("SipiPrefs", Context.MODE_PRIVATE)
52                   prefs.edit().apply {
53                       putString("last_email", email)
54                       putString("last_password", senha)
55                       apply()
56                   }
57                   Toast.makeText(this@LoginActivity, "Autenticado com sucesso!", Toast.LENGTH_SHORT).show()
58                   finish()
59               } else {
60                   // Falha
61                   Toast.makeText(this@LoginActivity, erro, Toast.LENGTH_LONG).show()
62               }
63           }
64       }
65   }
