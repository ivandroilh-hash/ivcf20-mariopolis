36       private val detailLauncher = registerForActivityResult(ActivityResultContracts.StartActivityForResult()) { result ->
37           if (result.resultCode == RESULT_OK) {
38               val nome = result.data?.getStringExtra("NOME")
39               val detalhamento = result.data?.getStringExtra("DETALHAMENTO") ?: ""
40               val condicoes = result.data?.getStringArrayListExtra("CONDICOES") ?: arrayListOf()
41               
42               moradoresNoImovel.find { it.imovel.nome == nome }?.apply {
43                   this.detalhamentoIndividual = detalhamento
44                   this.condicoesSaude = condicoes
45               }
46           }

63           adapter = MoradoresAdapter(moradoresNoImovel) { morador ->

96       private fun buscarMoradoresMesmoImovel(numeroImovel: String) {
97           val todos = ImovelData.imoveis
98           val filtrados = todos.filter { it.imovel == numeroImovel }
99           
100          moradoresNoImovel.clear()
101          
102          lifecycleScope.launch {
103              filtrados.forEach { imovel ->
104                  val historico = condicaoDao.getByNome(imovel.nome)
105                  val condicoesRaw = historico?.condicoes ?: imovel.condicoesIniciais
106                  val condicoesList = condicoesRaw.split(", ").filter { it.isNotEmpty() }
107                  
108                  moradoresNoImovel.add(MembroVisita(
109                      imovel = imovel,
110                      detalhamentoIndividual = "", 
111                      condicoesSaude = condicoesList
112                  ))
113              }
114              processarRegraPuerpera()
115              adapter.notifyDataSetChanged()
116          }
117      }
118  
119      private fun processarRegraPuerpera() {
120          val recemNascido = moradoresNoImovel.find { it.imovel.calcularDiasVida() < 45 }
121          if (recemNascido != null) {
122              moradoresNoImovel.find { it.imovel.nome == recemNascido.imovel.nomedamae }?.apply {
123                  if (!condicoesSaude.contains("Puérpera")) {
124                      val novaLista = condicoesSaude.toMutableList()
125                      novaLista.add("Puérpera")
126                      condicoesSaude = novaLista
127                  }
128              }
129          }
130      }
131  
132      private fun salvarVisitaColetiva() {
133          val principal = residentePrincipal
134          if (principal == null || moradoresNoImovel.isEmpty()) {
135              Toast.makeText(this, "Selecione um imóvel primeiro!", Toast.LENGTH_SHORT).show()
136              return
137          }
138  
139          val dataHora = SimpleDateFormat("dd/MM/yyyy HH:mm:ss", Locale("pt", "BR")).format(Date())
140  
141          lifecycleScope.launch {
142              try {
143                  val sb = StringBuilder()
144                  
145                  moradoresNoImovel.forEach { membro ->
146                      val statusStr = if (membro.presente) "Sim" else "Não"
147                      val condicoesFinais = membro.condicoesSaude.toMutableList()
148                      if (condicoesFinais.contains("Puérpera")) {
149                          condicoesFinais.remove("Gestante")
150                      }
151  
152                      condicaoDao.insertOrUpdate(PacienteCondicao(
153                          nome = membro.imovel.nome,
154                          detalhamento = "", 
155                          condicoes = condicoesFinais.joinToString(", ")
156                      ))
157  
158                      if (membro.imovel.nome != principal.nome) {
159                          sb.append("${membro.imovel.nome}\n")
160                      }
161                      
162                      sb.append("Imóvel: ${membro.imovel.imovel} | Código: ${membro.imovel.cod} | Presente: $statusStr\n")
163                      
164                      if (membro.imovel.dataNasc.isNotEmpty()) {
165                          sb.append("Data Nasc: ${membro.imovel.dataNasc}\n")
166                      }
167  
168                      if (membro.presente) {
169                          if (condicoesFinais.isNotEmpty()) {
170                              sb.append("Condições: ${condicoesFinais.joinToString(", ")}\n")
171                          }
172                          if (membro.detalhamentoIndividual.isNotEmpty()) {
173                              sb.append("Detalhamento: ${membro.detalhamentoIndividual}\n")
174                          }
175                      }
176                      sb.append("\n")
177                  }
178  
179                  val visita = Visita(
180                      imovel = principal.imovel,
181                      cod = principal.cod,
182                      nome = principal.nome,
183                      presente = true, 
184                      detalhamento = sb.toString().trimEnd(),
185                      dataHora = dataHora
186                  )
187                  dao.insert(visita)
188  
189                  Toast.makeText(this@VisitaActivity, "✅ Visita consolidada salva!", Toast.LENGTH_SHORT).show()
190                  finish()
191              } catch (e: Exception) {
192                  Toast.makeText(this@VisitaActivity, "❌ Erro: ${e.message}", Toast.LENGTH_LONG).show()
193              }
194          }
195      }

197      data class MembroVisita(
198          val imovel: Imovel,
199          var presente: Boolean = false,
200          var respondido: Boolean = false,
201          var detalhamentoIndividual: String = "",

205      inner class MoradoresAdapter(
206          private val lista: List<MembroVisita>,
207          private val onDetailClick: (MembroVisita) -> Unit
208      ) : RecyclerView.Adapter<MoradoresAdapter.ViewHolder>() {
209  
210          inner class ViewHolder(view: View) : RecyclerView.ViewHolder(view) {
211              val cbSim: CheckBox = view.findViewById(R.id.cbSim)
212              val cbNao: CheckBox = view.findViewById(R.id.cbNao)
213              val tvNome: TextView = view.findViewById(R.id.tvNomeResidente)
214              val btnDetalhes: ImageButton = view.findViewById(R.id.btnAbrirDetalhes)
215          }
216  
217          override fun onCreateViewHolder(parent: ViewGroup, viewType: Int): ViewHolder {
218              val v = LayoutInflater.from(parent.context).inflate(R.layout.item_residente_visita, parent, false)
219              return ViewHolder(v)
220          }
221  
222          override fun onBindViewHolder(holder: ViewHolder, position: Int) {
223              val item = lista[position]
224              holder.tvNome.text = item.imovel.nome
225              
226              holder.cbSim.setOnCheckedChangeListener(null)
227              holder.cbNao.setOnCheckedChangeListener(null)
228              
229              holder.cbSim.isChecked = item.presente && item.respondido
230              holder.cbNao.isChecked = !item.presente && item.respondido
231  
232              holder.cbSim.setOnClickListener {
233                  item.presente = true
234                  item.respondido = true
235                  notifyItemChanged(position)
236              }
237              
238              holder.cbNao.setOnClickListener {
239                  item.presente = false
240                  item.respondido = true
241                  notifyItemChanged(position)
242              }
243  
244              holder.btnDetalhes.setOnClickListener { onDetailClick(item) }
245          }
246  
247          override fun getItemCount() = lista.size
248      }
