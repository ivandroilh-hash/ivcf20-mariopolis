55       suspend fun autenticarComCredenciais(user: String, senha: String): String? {
56           // Se for e-mail (contém @), removemos apenas espaços. 
57           // Se for CPF (não contém @), removemos pontos e traços.
58           val userFinal = if (user.contains("@")) user.trim() else user.replace(Regex("[^0-9]"), "")
59           
60           return try {
61               Log.d(TAG, "Iniciando login para: $userFinal na URL: $BASE_URL")
62               val response = api.login(LoginRequest(username = userFinal, password = senha))
63               
64               if (response.isSuccessful) {
65                   val token = response.body()?.token
66                   if (!token.isNullOrBlank()) {
67                       authToken = "Token $token"
68                       Log.d(TAG, "Login realizado com sucesso")
69                       null 
70                   } else {
71                       "Erro: O servidor não enviou o token de acesso."
72                   }
73               } else {
74                   val code = response.code()
75                   val errorBody = response.errorBody()?.string()
76                   Log.e(TAG, "Erro no Login ($code): $errorBody")
77                   
78                   when (code) {
79                       400 -> "Erro 400: Dados de login inválidos ou campos faltando."
80                       401 -> "Usuário ou Senha incorretos."
81                       403 -> "Acesso negado ao servidor SIPI."
82                       404 -> "Erro: Endpoint de login não encontrado. Verifique a URL."
83                       else -> "Erro $code: Falha na autenticação."
84                   }
85               }
86           } catch (e: Exception) {
87               Log.e(TAG, "Exceção no Login: ${e.message}", e)
88               val msg = e.localizedMessage ?: e.message ?: "Erro desconhecido"
89               "Falha de conexão: $msg"
90           }
91       }

93       suspend fun sincronizar(response: QuestionnaireResponse, cnes: String): String {
94           if (authToken == null) return "Necessário realizar login primeiro."
95   
96           return try {
97               val token = authToken!!
98               val cpfIdosoLimpo = response.cpf.replace(Regex("[^0-9]"), "")
99               
100              // 1. Verifica ou Cadastra Idoso
101              val checkIdoso = api.buscarIdoso(token, cpfIdosoLimpo)
102              if (!checkIdoso.isSuccessful) {
103                  val idosoReq = mapearIdoso(response, cpfIdosoLimpo)
104                  val idosoResp = api.cadastrarIdoso(token, idosoReq)
105                  if (!idosoResp.isSuccessful) {
106                      val erroMsg = idosoResp.errorBody()?.string() ?: "Erro desconhecido"
107                      Log.e(TAG, "Falha no cadastro do idoso: $erroMsg")
108                      return "Erro Cadastro Idoso: $erroMsg"
109                  }
110              }
111  
112              // 2. Envia Avaliação
113              val avaliacaoReq = mapearAvaliacao(response, cpfIdosoLimpo, cnes)
114              val sendAval = api.enviarAvaliacao(token, avaliacaoReq)
115              
116              if (sendAval.isSuccessful) {
117                  "Sincronizado com sucesso!"
118              } else {
119                  val erroMsg = sendAval.errorBody()?.string() ?: "Erro desconhecido"
120                  Log.e(TAG, "Falha no envio da avaliação: $erroMsg")
121                  "Erro Envio Avaliação: $erroMsg"
122              }
123          } catch (e: Exception) {
124              Log.e(TAG, "Exceção na sincronização: ${e.message}")
125              "Erro de Conexão: ${e.message}"
126          }
127      }

129      private fun mapearIdoso(res: QuestionnaireResponse, cpf: String): IdosoCreateRequest {
130          val json = JSONObject(res.respostasJson)
131          return IdosoCreateRequest(
132              nome = res.nome, 
133              cpf = cpf, 
134              nascimento = converterData(res.dataNascimento),
135              sexo = if (res.sexo.startsWith("M", true)) "1" else "2",
136              municipio = "4115309", // Código de Mariópolis
137              cep = "85525000",
138              bairro = res.bairro.ifBlank { "Centro" }, 
139              rua = res.rua.ifBlank { "Principal" }, 
140              numero = res.numero.ifBlank { "SN" },
141              profissao = "Não informado",
142              escolaridade = when(json.optString("3")) {
143                  "Ensino Fundamental Incompleto" -> 1; "Ensino Fundamental Completo" -> 2
144                  "Ensino Médio Incompleto" -> 3; "Ensino Médio Completo" -> 4
145                  "Ensino Técnico Completo" -> 5; "Graduação incompleta" -> 6
146                  "Graduação Completa" -> 7; "Pós-Graduação Completa" -> 8
147                  "Não Alfabetizado" -> 9; else -> 10
148              },
149              estado_civil = when(json.optString("4")) {
150                  "Solteiro (a)" -> 1; "Casado (a)" -> 2; "Divorciado (a)" -> 3; "Viúvo (a)" -> 4; else -> 1
151              },
152              ilpi = 2, // Não
153              cuidador = 2 // Não
154          )
155      }

157      private fun mapearAvaliacao(res: QuestionnaireResponse, cpfIdoso: String, cnes: String): SipiAvaliacaoRequest {
158          val json = JSONObject(res.respostasJson)
159          fun toInt(q: String, defaultNao: Int = 2): Int = if (json.optString(q).startsWith("Sim", true)) 1 else defaultNao
160  
161          val pg24 = toInt("18") 
162          val imc = calcularImc(json)
163          val pg25 = if (imc > 0 && imc < 22) 1 else 2
164          val pantVal = json.optString("21").replace(",", ".").toDoubleOrNull()
165          val pg26 = if (pantVal != null && pantVal < 31) 1 else 2
166          val pg27 = toInt("22")
167          val pg14 = if (pg24 == 1 || pg25 == 1 || pg26 == 1 || pg27 == 1) 1 else 2
168  
169          val pg21 = toInt("28"); val pg22 = toInt("32"); val pg23 = toInt("33")
170          val pg20 = if (pg21 == 1 || pg22 == 1 || pg23 == 1) 1 else 2
171  
172          return SipiAvaliacaoRequest(
173              cpf_idoso = cpfIdoso,
174              cpf_usuario = CPF_PROFISSIONAL,
175              cnes = cnes,
176              created_at = converterData(res.dataHora.split(" ")[0]),
177              pg_1 = calcularFaixaEtaria(res.dataNascimento),
178              pg_2 = if (json.optString("6").contains("Excelente", true)) 1 else 2,
179              pg_3 = toInt("7"), pg_4 = toInt("8", 3), pg_5 = toInt("9", 4),
180              pg_6 = toInt("10"), pg_7 = toInt("11"), pg_8 = toInt("12"), pg_9 = toInt("13"),
181              pg_10 = toInt("14"), pg_11 = toInt("15"), pg_12 = toInt("16"), pg_13 = toInt("17"),
182              pg_14 = pg14, pg_24 = pg24, pg_25 = pg25, pg_26 = pg26, pg_27 = pg27,
183              pg_15 = toInt("23"), pg_16 = toInt("24"), pg_17 = toInt("25"),
184              pg_18 = toInt("26"), pg_19 = toInt("27"), pg_20 = pg20,
185              pg_21 = pg21, pg_22 = pg22, pg_23 = pg23,
186              status = 1,
187              score_total = 0,
188              dm_1 = 0, dm_2 = 0, dm_3 = 0, dm_4 = 0, dm_5 = 0, dm_6 = 0, dm_7 = 0, dm_8 = 0,
189              sub_dm_1 = 0, sub_dm_2 = 0, sub_dm_3 = 0, sub_dm_4 = 0, sub_dm_5 = 0, sub_dm_6 = 0, sub_dm_7 = 0, sub_dm_8 = 0
190          )
191      }

193      private fun converterData(data: String): String {
194          return try {
195              val date = SimpleDateFormat("dd/MM/yyyy", Locale.getDefault()).parse(data)
196              SimpleDateFormat("yyyy-MM-dd", Locale.getDefault()).format(date!!)
197          } catch (e: Exception) {
198              SimpleDateFormat("yyyy-MM-dd", Locale.getDefault()).format(Date())
199          }
200      }

202      private fun calcularFaixaEtaria(nasc: String): Int {
203          val idade = try {
204              val nascimento = Calendar.getInstance().apply { 
205                  time = SimpleDateFormat("dd/MM/yyyy", Locale.getDefault()).parse(nasc)!! 
206              }
207              val hoje = Calendar.getInstance()
208              var age = hoje.get(Calendar.YEAR) - nascimento.get(Calendar.YEAR)
209              if (hoje.get(Calendar.DAY_OF_YEAR) < nascimento.get(Calendar.DAY_OF_YEAR)) age--
210              age
211          } catch (e: Exception) { 60 }
212          return when { idade >= 85 -> 3; idade >= 75 -> 2; else -> 1 }
213      }

215      private fun calcularImc(json: JSONObject): Double {
216          val p = json.optString("19").replace(",", ".").toDoubleOrNull() ?: 0.0
217          val a = json.optString("20").replace(",", ".").toDoubleOrNull() ?: 0.0
218          return if (a > 0) p / (a * a) else 0.0
219      }
