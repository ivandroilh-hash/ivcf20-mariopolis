84       private suspend fun gerarListaConsolidada(): List<Any> {
85           val hoje = Calendar.getInstance()
86           val limite = Calendar.getInstance().apply { add(Calendar.DAY_OF_YEAR, 15) }
87           val sdf = SimpleDateFormat("dd/MM/yyyy", Locale.getDefault())
88   
89           val gestantes = mutableListOf<AlertaItem>()
90           val puerperas = mutableListOf<AlertaItem>()
91           val criancas = mutableListOf<AlertaItem>()
92           val idosos = mutableListOf<AlertaItem>()
93   
94           val moradoresDoAcs = ImovelData.imoveis.filter { 
95               it.acs.trim().equals(acsSelecionado, ignoreCase = true) 
96           }
97   
98           val mãesDeRecemNascidos = moradoresDoAcs.filter { it.calcularDiasVida() < 45 }.map { it.imovel to it.nomedamae }
99   
100          moradoresDoAcs.forEach { imovel ->
101              // Busca por nome exato ou contendo parte do nome para evitar problemas de espaço ou nomes incompletos
102              val historico = condicaoDao.getByNome(imovel.nome.trim())
103              val condicoesNoBanco = historico?.condicoes ?: ""
104              val todasCondicoes = (imovel.condicoesIniciais + ", " + condicoesNoBanco).lowercase()
105              
106              val agendamentoData = historico?.agendamentoData
107              val agendamentoProf = historico?.agendamentoProfissional
108  
109              if (todasCondicoes.contains("gestante")) {
110                  gestantes.add(AlertaItem(imovel.nome, "Gestante", imovel.bairro, "", "", "Gestante", agendamentoData, agendamentoProf))
111              }
112  
113              val eMãeDeBebeNovo = mãesDeRecemNascidos.any { it.first == imovel.imovel && it.second == imovel.nome }
114              if (todasCondicoes.contains("puerpera") || todasCondicoes.contains("puérpera") || eMãeDeBebeNovo) {
115                  puerperas.add(AlertaItem(imovel.nome, "Puérpera", imovel.bairro, "", "", "Puérpera", agendamentoData, agendamentoProf))
116              }
117  
118              if (imovel.dataNasc.isEmpty()) return@forEach
119              
120              try {
121                  val dataNasc = sdf.parse(imovel.dataNasc) ?: return@forEach
122                  for (mes in listOf(1, 2, 3, 4, 5, 6, 7, 9)) {
123                      val dataMarco = Calendar.getInstance().apply { time = dataNasc; add(Calendar.MONTH, mes) }
124                      if (estaNoIntervalo(dataMarco, hoje, limite)) {
125                          criancas.add(AlertaItem(imovel.nome, "Completará $mes meses", imovel.bairro, sdf.format(dataMarco.time), imovel.nomedamae, "Criança", agendamentoData, agendamentoProf))
126                      }
127                  }
128                  val dezoitoMeses = Calendar.getInstance().apply { time = dataNasc; add(Calendar.MONTH, 18) }
129                  if (estaNoIntervalo(dezoitoMeses, hoje, limite)) criancas.add(AlertaItem(imovel.nome, "Completará 1 ano e 6 meses", imovel.bairro, sdf.format(dezoitoMeses.time), imovel.nomedamae, "Criança", agendamentoData, agendamentoProf))
130                  for (ano in listOf(1, 2)) {
131                      val dataAno = Calendar.getInstance().apply { time = dataNasc; add(Calendar.YEAR, ano) }
132                      if (estaNoIntervalo(dataAno, hoje, limite)) {
133                          val textoAno = if (ano == 1) "1 ano" else "$ano anos"
134                          criancas.add(AlertaItem(imovel.nome, "Completará $textoAno", imovel.bairro, sdf.format(dataAno.time), imovel.nomedamae, "Criança", agendamentoData, agendamentoProf))
135                      }
136                  }
137                  val sessentaAnos = Calendar.getInstance().apply { time = dataNasc; add(Calendar.YEAR, 60) }
138                  if (estaNoIntervalo(sessentaAnos, hoje, limite)) {
139                      idosos.add(AlertaItem(imovel.nome, "Completará 60 anos", imovel.bairro, sdf.format(sessentaAnos.time), "", "Idoso", agendamentoData, agendamentoProf))
140                  }
141              } catch (e: Exception) { }
142          }
143  
144          val listaFinal = mutableListOf<Any>()
145          if (gestantes.isNotEmpty()) { listaFinal.add(" GESTANTES "); listaFinal.addAll(gestantes.distinctBy { it.nome }.sortedBy { it.bairro }) }
146          if (puerperas.isNotEmpty()) { listaFinal.add(" PUÉRPERAS "); listaFinal.addAll(puerperas.distinctBy { it.nome }.sortedBy { it.bairro }) }
147          if (criancas.isNotEmpty()) { listaFinal.add(" CRIANÇAS (MARCOS DE IDADE) "); listaFinal.addAll(criancas.sortedWith(compareBy({ it.bairro }, { it.dataEvento }))) }
148          if (idosos.isNotEmpty()) { listaFinal.add(" IDOSOS (COMPLETANDO 60 ANOS) "); listaFinal.addAll(idosos.sortedBy { it.bairro }) }
149          return listaFinal
150      }

152      private fun mostrarDialogAgendamento(item: AlertaItem) {
153          if (item.tipo == "Idoso") return
154          val builder = AlertDialog.Builder(this, R.style.CustomAlertDialogTheme)
155          builder.setTitle("Agendar para: ${item.nome}")
156          val layout = LinearLayout(this).apply {
157              orientation = LinearLayout.VERTICAL
158              setPadding(50, 40, 50, 10)
159              setBackgroundColor(Color.WHITE)
160          }
161          val etData = EditText(this).apply {
162              hint = "Data do Agendamento"
163              setText(item.agendamentoData ?: "")
164              isFocusable = false
165              setTextColor(Color.BLACK)
166              setHintTextColor(Color.GRAY)
167              setOnClickListener {
168                  val cal = Calendar.getInstance()
169                  // Usando o tema definido no styles.xml para o calendário
170                  DatePickerDialog(this@AlertasActivity, R.style.TemaCalendario, { _, y, m, d ->
171                      setText(String.format(Locale.getDefault(), "%02d/%02d/%d", d, m + 1, y))
172                  }, cal.get(Calendar.YEAR), cal.get(Calendar.MONTH), cal.get(Calendar.DAY_OF_MONTH)).show()
173              }
174          }
175          layout.addView(etData)
176          val spinnerProf = Spinner(this)
177          val profissionais = when (item.tipo) {
178              "Criança" -> listOf("Selecione o Profissional", "Dentista", "Enfermeiro(a)", "Fisioterapeuta", "Médico(a)", "Nutricionista")
179              "Gestante", "Puérpera" -> listOf("Selecione o Profissional", "Enfermeiro(a)", "Médico(a)")
180              else -> null
181          }
182          profissionais?.let { list ->
183              val adapter = object : ArrayAdapter<String>(this, android.R.layout.simple_spinner_item, list) {
184                  override fun getView(position: Int, convertView: View?, parent: ViewGroup): View {
185                      val v = super.getView(position, convertView, parent) as TextView
186                      v.setTextColor(Color.BLACK)
187                      return v
188                  }
189              }
190              adapter.setDropDownViewResource(android.R.layout.simple_spinner_dropdown_item)
191              spinnerProf.adapter = adapter
192              item.agendamentoProf?.let { currentProf ->
193                  val pos = list.indexOf(currentProf)
194                  if (pos >= 0) spinnerProf.setSelection(pos)
195              }
196              layout.addView(spinnerProf)
197          }
198          builder.setView(layout)
199          builder.setPositiveButton("Salvar") { _, _ ->
200              val data = etData.text.toString()
201              val prof = if (profissionais != null) spinnerProf.selectedItem.toString() else null
202              if (data.isNotEmpty()) {
203                  lifecycleScope.launch {
204                      val pc = condicaoDao.getByNome(item.nome.trim()) ?: PacienteCondicao(item.nome.trim(), "", "")
205                      val updatedPc = pc.copy(agendamentoData = data, agendamentoProfissional = if (prof != "Selecione o Profissional") prof else null)
206                      withContext(Dispatchers.IO) { condicaoDao.insertOrUpdate(updatedPc) }
207                      atualizarLista()
208                      Toast.makeText(this@AlertasActivity, "Agendamento salvo!", Toast.LENGTH_SHORT).show()
209                  }
210              }
211          }
212          builder.setNegativeButton("Cancelar", null)
213          builder.show()
214      }

216      private fun gerarPdfAlertas(lista: List<Any>) {
217          val pdfDocument = PdfDocument()
218          val paint = Paint().apply { textSize = 12f }
219          val paintBold = Paint().apply { isFakeBoldText = true; textSize = 12f }
220          val paintHeader = Paint().apply { isFakeBoldText = true; textSize = 14f; color = Color.BLUE }
221          var pageNumber = 1
222          var pageInfo = PdfDocument.PageInfo.Builder(595, 842, pageNumber).create()
223          var page = pdfDocument.startPage(pageInfo)
224          var canvas = page.canvas
225          var y = 50f
226          canvas.drawText("Alertas de Visitas (15 dias) - ACS: $acsSelecionado", 50f, y, paintHeader)
227          y += 40f
228          lista.forEach { item ->
229              if (y > 750) {
230                  pdfDocument.finishPage(page)
231                  pageNumber++
232                  pageInfo = PdfDocument.PageInfo.Builder(595, 842, pageNumber).create()
233                  page = pdfDocument.startPage(pageInfo)
234                  canvas = page.canvas
235                  y = 50f
236              }
237              if (item is String) {
238                  y += 10f
239                  canvas.drawText(item, 50f, y, paintHeader)
240                  y += 25f
241              } else if (item is AlertaItem) {
242                  canvas.drawText("${item.nome} - Bairro: ${item.bairro}", 50f, y, paintBold)
243                  y += 18f
244                  val infoAdicional = if (item.dataEvento.isNotEmpty()) "${item.marco} em ${item.dataEvento}" else item.marco
245                  canvas.drawText(infoAdicional, 70f, y, paint)
246                  y += 18f
247                  if (item.mae.isNotEmpty()) {
248                      canvas.drawText("Mãe: ${item.mae}", 70f, y, paint)
249                      y += 18f
250                  }
251                  if (!item.agendamentoData.isNullOrEmpty()) {
252                      val textoAgend = "📅 Agendado: ${item.agendamentoData}" + (if (!item.agendamentoProf.isNullOrEmpty()) " (${item.agendamentoProf})" else "")
253                      canvas.drawText(textoAgend, 70f, y, paintBold)
254                      y += 18f
255                  }
256                  y += 10f
257              }
258          }
259          pdfDocument.finishPage(page)
260          val timeStamp = SimpleDateFormat("dd-MM-yyyy_HH-mm", Locale.getDefault()).format(Date())
261          val fileName = "Alertas_15dias_${acsSelecionado}_$timeStamp.pdf"
262          try {
263              if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.Q) {
264                  val contentValues = ContentValues().apply {
265                      put(MediaStore.MediaColumns.DISPLAY_NAME, fileName)
266                      put(MediaStore.MediaColumns.MIME_TYPE, "application/pdf")
267                      put(MediaStore.MediaColumns.RELATIVE_PATH, Environment.DIRECTORY_DOWNLOADS)
268                      put(MediaStore.MediaColumns.IS_PENDING, 1)
269                  }
270                  val uri = contentResolver.insert(MediaStore.Downloads.EXTERNAL_CONTENT_URI, contentValues)
271                  uri?.let {
272                      contentResolver.openOutputStream(it)?.use { pdfDocument.writeTo(it) }
273                      contentValues.clear(); contentValues.put(MediaStore.MediaColumns.IS_PENDING, 0)
274                      contentResolver.update(it, contentValues, null, null)
275                  }
276              } else {
277                  val file = File(Environment.getExternalStoragePublicDirectory(Environment.DIRECTORY_DOWNLOADS), fileName)
278                  pdfDocument.writeTo(FileOutputStream(file))
279              }
280              Toast.makeText(this, "✅ PDF salvo em Downloads!", Toast.LENGTH_LONG).show()
281          } catch (e: Exception) { Toast.makeText(this, "❌ Erro: ${e.message}", Toast.LENGTH_LONG).show() }
282          finally { pdfDocument.close() }
283      }

285      private fun estaNoIntervalo(alvo: Calendar, inicio: Calendar, fim: Calendar): Boolean {
286          return !alvo.before(inicio) && !alvo.after(fim)
287      }

291      inner class AlertasAdapter(private val lista: List<Any>) : RecyclerView.Adapter<RecyclerView.ViewHolder>() {
292          private val TYPE_HEADER = 0
293          private val TYPE_ITEM = 1
294          override fun getItemViewType(position: Int): Int = if (lista[position] is String) TYPE_HEADER else TYPE_ITEM
295          override fun onCreateViewHolder(parent: ViewGroup, viewType: Int): RecyclerView.ViewHolder {
296              return if (viewType == TYPE_HEADER) {
297                  HeaderViewHolder(LayoutInflater.from(parent.context).inflate(android.R.layout.simple_list_item_1, parent, false))
298              } else {
299                  ItemViewHolder(LayoutInflater.from(parent.context).inflate(R.layout.item_alerta, parent, false))
300              }
301          }
302          override fun onBindViewHolder(holder: RecyclerView.ViewHolder, position: Int) {
303              if (holder is HeaderViewHolder) {
304                  holder.tv.text = lista[position] as String
305                  holder.tv.setBackgroundColor(Color.parseColor("#E3F2FD"))
306                  holder.tv.setTextColor(Color.parseColor("#1976D2"))
307                  holder.tv.setTypeface(null, Typeface.BOLD)
308              } else if (holder is ItemViewHolder) {
309                  val item = lista[position] as AlertaItem
310                  holder.itemView.setBackgroundColor(Color.WHITE)
311                  holder.tvNome.text = "${item.nome} - Bairro: ${item.bairro}"
312                  holder.tvNome.setTextColor(Color.BLACK)
313                  val txt = if (item.dataEvento.isNotEmpty()) "${item.marco} em ${item.dataEvento}" else item.marco
314                  var txtFinal = if (item.mae.isNotEmpty()) "$txt\nMãe: ${item.mae}" else txt
315                  if (!item.agendamentoData.isNullOrEmpty()) {
316                      txtFinal += "\n📅 Agendado para: ${item.agendamentoData}"
317                      if (!item.agendamentoProf.isNullOrEmpty()) { txtFinal += " (${item.agendamentoProf})" }
318                  }
319                  holder.tvDetalhes.text = txtFinal
320                  holder.tvDetalhes.setTextColor(Color.DKGRAY)
321                  holder.itemView.setOnClickListener { mostrarDialogAgendamento(item) }
322              }
323          }
324          override fun getItemCount() = lista.size
325          inner class HeaderViewHolder(v: View) : RecyclerView.ViewHolder(v) { val tv = v.findViewById<TextView>(android.R.id.text1) }
326          inner class ItemViewHolder(v: View) : RecyclerView.ViewHolder(v) {
327              val tvNome: TextView = v.findViewById(R.id.tvNomeBairro)
328              val tvDetalhes: TextView = v.findViewById(R.id.tvDetalhesAlerta)
329          }
330      }
