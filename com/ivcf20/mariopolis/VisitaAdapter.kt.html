1    package com.ivcf20.mariopolis
2    
3    import android.view.LayoutInflater
4    import android.view.View
5    import android.view.ViewGroup
6    import androidx.recyclerview.widget.RecyclerView
7    import com.ivcf20.mariopolis.databinding.ItemVisitaBinding
8    
9    class VisitadaAdapter(
10       private var lista: List<Visita>,
11       private val onSelectionChanged: (Set<Long>) -> Unit,
12       private val onItemClick: (Visita) -> Unit = {}
13   ) : RecyclerView.Adapter<VisitadaAdapter.ViewHolder>() {
14   
15       private val selectedIds = mutableSetOf<Long>()
16   
17       class ViewHolder(val binding: ItemVisitaBinding) : RecyclerView.ViewHolder(binding.root)
18   
19       override fun onCreateViewHolder(parent: ViewGroup, viewType: Int): ViewHolder {
20           val binding = ItemVisitaBinding.inflate(LayoutInflater.from(parent.context), parent, false)
21           return ViewHolder(binding)
22       }
23   
24       override fun onBindViewHolder(holder: ViewHolder, position: Int) {
25           val item = lista[position]
26           holder.binding.tvNome.text = item.nome
27           holder.binding.tvDataHora.text = item.dataHora
28   
29           holder.binding.cbItem.visibility = View.VISIBLE
30           holder.binding.cbItem.setOnCheckedChangeListener(null)
31           holder.binding.cbItem.isChecked = selectedIds.contains(item.id)
32   
33           holder.binding.cbItem.setOnCheckedChangeListener { _, isChecked ->
34               if (isChecked) selectedIds.add(item.id) else selectedIds.remove(item.id)
35               onSelectionChanged(selectedIds)
36           }
37   
38           holder.itemView.setOnClickListener {
39               onItemClick(item)
40           }
41       }
42   
43       fun selectAll(select: Boolean) {
44           if (select) {
45               selectedIds.addAll(lista.map { it.id })
46           } else {
47               selectedIds.clear()
48           }
49           notifyDataSetChanged()
50           onSelectionChanged(selectedIds)
51       }
52   
53       fun getSelectedIds() = selectedIds.toList()
54   
55       fun updateData(novaLista: List<Visita>) {
56           lista = novaLista
57           notifyDataSetChanged()
58       }
59   
60       override fun getItemCount() = lista.size
61   }
