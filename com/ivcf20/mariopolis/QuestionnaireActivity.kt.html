186      private fun calcularImcAtual(): Double {
187          val p = respostasTexto[19]?.replace(",", ".")?.toDoubleOrNull() ?: 0.0
188          val a = respostasTexto[20]?.replace(",", ".")?.toDoubleOrNull() ?: 0.0
189          if (a <= 0) return 0.0
190          val imc = p / (a * a); val integerPart = Math.floor(imc); val cents = Math.round((imc - integerPart) * 100).toInt()
191          return if (cents <= 50) integerPart else Math.ceil(imc)
192      }

194      private fun salvarRespostaAtual() {
195          val q = questions[currentIndex]; val prefs = getSharedPreferences("VisitasPrefs", MODE_PRIVATE); val editor = prefs.edit()
196          if (q.numero == 19) {
197              respostasTexto[19] = editTextPeso?.text.toString().trim()
198              respostasTexto[20] = editTextAltura?.text.toString().trim()
199              editor.putString("q19_resposta", respostasTexto[19]); editor.putString("q20_resposta", respostasTexto[20])
200          } else if (q.multiSelect && checkboxesAtuais.isNotEmpty()) {
201              val sel = checkboxesAtuais.filter { it.isChecked }.joinToString("; ") { it.text.toString() }
202              if (sel.isNotEmpty()) respostasRadio[q.numero] = sel else respostasRadio.remove(q.numero)
203          } else if (spinnerAtual != null) { respostasRadio[q.numero] = spinnerAtual!!.selectedItem.toString() }
204          else { radioGroupAtual?.let { rg -> val id = rg.checkedRadioButtonId; if (id != -1) respostasRadio[q.numero] = rg.findViewById<RadioButton>(id).text.toString() } }
205  
206          if (q.numero == 11 && respostasRadio[11] == "Não") { editor.putString("q12_resposta", "Não"); editor.putString("q13_resposta", "Não") }
207          if (q.numero == 29 || q.numero == 30 || q.numero == 31) {
208              val c29 = (respostasRadio[29] ?: "").split("; ").filter { it.isNotBlank() && it != "Sem resposta" }.size
209              val c30 = (respostasRadio[30] ?: "").split("; ").filter { it.isNotBlank() && it != "Sem resposta" }.size
210              val c31 = (respostasRadio[31] ?: "").split("; ").filter { it.isNotBlank() && it != "Sem resposta" }.size
211              editor.putString("q28_resposta", if (c29 + c30 + c31 >= 5) "Sim" else "Não")
212          }
213          editTextAtual?.let { val t = it.text.toString().trim(); if (t.isNotEmpty()) respostasTexto[q.numero] = t else respostasTexto.remove(q.numero) }
214          if (q.numero == 1) {
215              val nomeAtual = respostasTexto[1] ?: ""
216              if (prefs.getBoolean("p_selected", false) && nomeAtual != prefs.getString("q1_nome_selecionado", "")) {
217                  editor.remove("p_cpf").remove("p_sexo").remove("p_cns").remove("p_rua").remove("p_numero").remove("p_bairro").remove("p_selected")
218              }
219              editor.putString("q1_nome_selecionado", nomeAtual)
220          }
221          if (q.numero != 19) {
222              val i = q.numero
223              val resp = when {
224                  respostasRadio[i] != null && respostasTexto[i] != null -> "${respostasRadio[i]} | ${respostasTexto[i]}"
225                  respostasRadio[i] != null -> respostasRadio[i]!!
226                  respostasTexto[i] != null -> respostasTexto[i]!!
227                  else -> "Sem resposta"
228              }
229              editor.putString("q${i}_resposta", resp)
230          }
231          editor.apply()
232      }

234      private fun proximoOuSalvar() {
235          salvarRespostaAtual()
236          if (currentIndex < questions.size - 1) {
237              val q = questions[currentIndex]
238              if (q.numero == 11 && respostasRadio[11] == "Não") { val idx14 = questions.indexOfFirst { it.numero == 14 }; currentIndex = if (idx14 != -1) idx14 else currentIndex + 1 }
239              else if (q.numero == 19) currentIndex += 2 else currentIndex++
240              carregarPergunta()
241          } else finalizarQuestionario()
242      }

244      private fun finalizarQuestionario() {
245          val prefs = getSharedPreferences("VisitasPrefs", MODE_PRIVATE); val nome = prefs.getString("q1_resposta", "Sem Nome") ?: "Sem Nome"
246          val dataHora = SimpleDateFormat("dd/MM/yyyy HH:mm:ss", Locale.getDefault()).format(Date())
247          val json = JSONObject(); questions.forEach { json.put(it.numero.toString(), prefs.getString("q${it.numero}_resposta", "Sem resposta")) }
248          
249          var cpf = ""; var sexo = ""; var dataNasc = ""; var cns = ""; var rua = ""; var numero = ""; var bairro = ""
250          
251          if (prefs.getBoolean("p_selected", false)) {
252              cpf = prefs.getString("p_cpf", "") ?: ""
253              sexo = prefs.getString("p_sexo", "") ?: ""
254              dataNasc = prefs.getString("q2_resposta", "") ?: ""
255              cns = prefs.getString("p_cns", "") ?: ""
256              rua = prefs.getString("p_rua", "") ?: ""
257              numero = prefs.getString("p_numero", "") ?: ""
258              bairro = prefs.getString("p_bairro", "") ?: ""
259              
260              json.put("p_cpf", cpf); json.put("p_sexo", sexo)
261              json.put("p_cns", cns); json.put("p_rua", rua)
262              json.put("p_numero", numero); json.put("p_bairro", bairro)
263              json.put("p_selected", true)
264          }
265  
266          lifecycleScope.launch {
267              try {
268                  val response = QuestionnaireResponse(
269                      nome = nome,
270                      cpf = cpf,
271                      sexo = sexo,
272                      dataNascimento = dataNasc,
273                      cns = cns,
274                      rua = rua,
275                      numero = numero,
276                      bairro = bairro,
277                      dataHora = dataHora,
278                      respostasJson = json.toString()
279                  )
280                  dao.insert(response)
281                  val fontSize = prefs.getFloat("current_text_size", 22f); prefs.edit().clear().putFloat("current_text_size", fontSize).apply()
282                  Toast.makeText(this@QuestionnaireActivity, "✅ Salvo!", Toast.LENGTH_LONG).show(); finish()
283              } catch (e: Exception) { Toast.makeText(this@QuestionnaireActivity, "❌ Erro: ${e.message}", Toast.LENGTH_LONG).show() }
284          }
285      }
